1//! Types and other definitions for defining Amimono applications.
2//!
3//! It's important that [`AppConfig`]s be constructed deterministically so that
4//! all workloads can use the same value. In the future this may be encouraged
5//! by allowing configuration code to be written with `const fn`s, but this is
6//! not currently implemented.
7//!
8//! # Example
9//!
10//! Most config values are constructed with builders such as [`AppBuilder`] and
11//! [`JobBuilder`], however [`ComponentConfig`] is meant to be constructed
12//! directly. The suggested way to organize your configuration code is to have
13//! one function per component that returns a [`ComponentConfig`], and then a
14//! single function close to your `main()` function that assembles these into
15//! an [`AppConfig`]. This might look as follows:
16//!
17//! ```
18//! use amimono::config::{AppBuilder, AppConfig, JobBuilder};
19//!
20//! use crate::backend;
21//! use crate::frontend;
22//!
23//! pub fn configure() -> AppConfig {
24//!     AppBuilder::new()
25//!         .add_job(
26//!             JobBuilder::new()
27//!                 .with_label("backend")
28//!                 .add_component(backend::emailer::component())
29//!                 .add_component(backend::accounts::component())
30//!                 .add_component(backend::images::component())
31//!                 .add_component(backend::orders::component())
32//!                 .build()
33//!         )
34//!         .add_job(
35//!             JobBuilder::new()
36//!                 .with_label("cache")
37//!                 .add_component(backend::cache::component())
38//!                 .build()
39//!         )
40//!         .add_job(
41//!             JobBuilder::new()
42//!                 .with_label("frontend")
43//!                 .add_component(frontend::component())
44//!                 .build()
45//!         )
46//!         .build()
47//! }
48//! ```
49//!
50//! However you are free to organize things in whatever way you prefer.

72/// The configuration for a single component.
73pub struct ComponentConfig {
74    /// This component's label, a string identifier. Every component must have a
75    /// unique label. The label is mostly used for external things like logging
76    /// and as a key in config files. Within the application, components are
77    /// identified with a type that implements the `Component` trait.
78    pub label: String,
79
80    /// An opaque identifier for this component's `Component` impl. This can
81    /// be generated with `Component::id()`. A `Component` impl is necessary
82    /// for accessing information such as bindings.
83    pub id: ComponentId,
84
85    /// The binding type requested by this component. When the component starts,
86    /// the allocated binding can be accessed with
87    /// [`runtime::binding`](crate::runtime::binding)
88    pub binding: BindingType,
89
90    /// The component's entry point.
91    pub entry: fn(),
92}

94/// A fully configured application.
95///
96/// Refer to the [module-level documentation][crate::config] for more information.
97pub struct AppConfig {
98    component_jobs: HashMap<String, String>,
99    jobs: HashMap<String, JobConfig>,
100}
101
102impl AppConfig {
103    pub fn component_job(&self, label: &str) -> Option<&str> {
104        self.component_jobs.get(label).map(|s| s.as_str())
105    }
106
107    /// A function to retrieve a `JobConfig` by its label.
108    pub fn job(&self, label: &str) -> Option<&JobConfig> {
109        self.jobs.get(label)
110    }
111
112    /// An iterator over the `JobConfig`s in the application.
113    pub fn jobs(&self) -> impl Iterator<Item = &JobConfig> {
114        self.jobs.values()
115    }
116}

118/// A fully configured job.

124}
125
126impl JobConfig {
127    /// An iterator over the `ComponentConfig`s in the job.
128    pub fn components(&self) -> impl Iterator<Item = &ComponentConfig> {
129        self.components.values()
130    }
131
132    /// The job's label.
133    pub fn label(&self) -> &str {

138/// A helper for constructing a `JobConfig`.
139///
140/// Refer to the [module-level documentation][crate::config] for more information.
141pub struct JobBuilder {
142    label: Option<String>,
143    components: HashMap<String, ComponentConfig>,
144}
145
146impl JobBuilder {
147    /// Create an empty `JobBuilder`.
148    pub fn new() -> JobBuilder {
149        JobBuilder {
150            label: None,
151            components: HashMap::new(),
152        }
153    }
154
155    /// Convert the builder into a `JobConfig`.
156    pub fn build(self) -> JobConfig {
157        let comps = self.components;
158        if comps.len() == 0 {
159            panic!("jobs must have at least one component");
160        }
161        let label = match self.label {
162            Some(label) => label,
163            None => {
164                if comps.len() > 1 {
165                    panic!("jobs with multiple components must have an explicit label")
166                }
167                comps.values().next().unwrap().label.clone()
168            }
169        };
170        JobConfig {
171            label,
172            components: comps,
173        }
174    }
175
176    /// Set the job's label.
177    pub fn with_label<S: Into<String>>(mut self, label: S) -> JobBuilder {
178        self.label = Some(label.into());
179        self
180    }
181
182    /// Add a component to the job.
183    pub fn add_component<C: Into<ComponentConfig>>(mut self, comp: C) -> JobBuilder {
184        let comp = comp.into();
185        let key = comp.label.clone();
186        if self.components.insert(key.clone(), comp).is_some() {
187            panic!("duplicate component label: {}", key);
188        }
189        self
190    }
191}
192
193impl From<JobBuilder> for JobConfig {
194    fn from(builder: JobBuilder) -> Self {
195        builder.build()
196    }
197}
198
199impl From<ComponentConfig> for JobConfig {
200    fn from(value: ComponentConfig) -> Self {
201        JobBuilder::new().add_component(value).build()
202    }
203}
204
205impl From<AppBuilder> for AppConfig {
206    fn from(builder: AppBuilder) -> Self {
207        builder.build()
208    }
209}

211/// A helper for constructing an `AppConfig`.
212///
213/// Refer to the [module-level documentation][crate::config] for more information.
214pub struct AppBuilder {
215    app: AppConfig,
216}
217
218impl AppBuilder {
219    /// Create an empty `AppBuilder`.
220    pub fn new() -> AppBuilder {
221        AppBuilder {
222            app: AppConfig {
223                component_jobs: HashMap::new(),
224                jobs: HashMap::new(),
225            },
226        }
227    }
228
229    /// Convert the builder into an `AppConfig`.
230    pub fn build(self) -> AppConfig {
231        self.app
232    }
233
234    /// Add a job to the app.
235    pub fn add_job<J: Into<JobConfig>>(mut self, job: J) -> AppBuilder {
236        let job = job.into();
237        let label = job.label.clone();
238        for comp in job.components() {
239            let comp_label = comp.label.clone();
240            let current_job = self
241                .app
242                .component_jobs
243                .insert(comp.label.clone(), label.clone());
244            if let Some(other_label) = current_job {
245                panic!(
246                    "component {} already assigned to job {}",
247                    comp_label, other_label
248                );
249            }
250        }
251        if self.app.jobs.insert(label.clone(), job).is_some() {
252            panic!("duplicate job label: {}", label);
253        }
254        self
255    }
