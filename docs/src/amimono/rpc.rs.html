1//! Subsystem for building components with an RPC interface.
2//!
3//! It's recommended to use this module via the [`rpc_ops!`][crate::rpc_ops]
4//! macro. When using that macro, it's rarely necessary to use any of the
5//! definitions in this module directly, however they are documented for the
6//! sake of completeness.

21/// A value that can be used as an RPC request or response.
22///
23/// Message types created by the [`rpc_ops!`][crate::rpc_ops] macro are
24/// automatically given an `RpcMessage` impl.
25pub trait RpcMessage: Serialize + for<'a> Deserialize<'a> + Send + 'static {
26    fn verb(&self) -> &'static str;
27}
28
29/// A value that can be used as an RPC request handler.
30///
31/// The `Instance` struct created by the [`rpc_ops!`][crate::rpc_ops] macro is
32/// automatically given an `Rpc` impl.
33pub trait Rpc: Sync + Send + 'static {
34    type Request: RpcMessage;
35    type Response: RpcMessage;
36
37    fn start() -> impl Future<Output = Self> + Send;
38
39    fn handle(&self, q: Self::Request) -> impl Future<Output = Self::Response> + Send;
40}
41
42type RpcInstance<R> = Arc<SetOnce<R>>;
43
44/// An RPC component, parameterized by an `Rpc` impl.
45///
46/// This type is an implementation detail that you should not need to use
47/// directly. The `ComponentConfig` returned by [`component`] is keyed by this
48/// type.
49pub struct RpcComponent<R>(PhantomData<R>);
50
51impl<R: Rpc> Component for RpcComponent<R> {
52    type Instance = RpcInstance<R>;
53}
54
55impl<R: Rpc> RpcComponent<R> {
56    #[tokio::main]
57    async fn entry() {
58        let instance = Arc::new(SetOnce::new());
59        runtime::set_instance::<Self>(instance.clone());
60        // we must call set_instance() asap, because get_instance::<T> blocks
61        // until the corresponding set_instance::<T> is called and we do not
62        // want to block in start() impls that make RPC calls.
63        instance.set(R::start().await).ok().unwrap();
64        rpc_http_server::<R>(instance).await;
65        panic!("rpc_http_server exited");
66    }
67
68    fn component(label: String) -> ComponentConfig {
69        ComponentConfig {
70            label,
71            id: RpcComponent::<R>::id(),
72            binding: BindingType::Http,
73            entry: Self::entry,
74        }
75    }
76}

78/// Create a `ComponentConfig` for an `Rpc` impl.
79///
80/// When using the [`rpc_ops!`][crate::rpc_ops] macro, you should instead use
81/// the `component` function defined by that macro.
82pub fn component<R: Rpc>(label: String) -> ComponentConfig {
83    RpcComponent::<R>::component(label)
84}

86/// An error when making an RPC call.
87#[derive(Debug, Clone)]
88pub enum RpcError {
89    /// A miscellaneous error with an unstructured string message. These should
90    /// generally be assumed to be unrecoverable.
91    Misc(String),
92}
93
94/// A client for making requests to an RPC component.
95///
96/// Cloning values of this type will result in clients that share resources
97/// such as connection pools.
98///
99/// The `Client` struct defined by the [`rpc_ops!`][crate::rpc_ops] macro is a
100/// thin wrapper around this type.
101pub enum RpcClient<R> {
102    Local { inner: LazyLock<RpcInstance<R>> },
103    Http { inner: RpcHttpClient<R> },
104}
105
106impl<R: Rpc> Clone for RpcClient<R> {
107    fn clone(&self) -> Self {
108        match self {
109            RpcClient::Local { .. } => RpcClient::new(),
110            RpcClient::Http { inner } => RpcClient::Http {
111                inner: inner.clone(),
112            },
113        }
114    }
115}
116
117impl<R: Rpc> RpcClient<R> {
118    /// Create a new client for a particular `Rpc` impl. If an existing client
119    /// can be cloned, that should be preferred, as it will result in resources
120    /// being shared between the clients.
121    pub fn new() -> RpcClient<R> {
122        if runtime::is_local::<RpcComponent<R>>() {
123            RpcClient::Local {
124                inner: LazyLock::new(|| runtime::get_instance::<RpcComponent<R>>().clone()),
125            }
126        } else {
127            RpcClient::Http {
128                inner: RpcHttpClient::new(),
129            }
130        }
131    }
132
133    /// Send a request. If the target `Rpc` impl belongs to a component that is
134    /// running in the same process, this will result in the target handler
135    /// being invoked directly.
136    pub async fn call(&self, q: R::Request) -> Result<R::Response, RpcError> {
137        match self {
138            RpcClient::Local { inner } => Ok(inner.wait().await.handle(q).await),
139            RpcClient::Http { inner } => inner.call(q).await,
140        }
141    }
142
143    /// Returns a reference to the underlying `Rpc` impl if the target component
144    /// is running in the same process.
145    pub async fn local(&self) -> Option<&R> {
146        match self {
147            RpcClient::Local { inner } => Some(inner.wait().await),
148            _ => None,
149        }
150    }
151}

153async fn rpc_http_server<R: Rpc>(inner: RpcInstance<R>) {
154    let label = runtime::label::<RpcComponent<R>>();
155    let (addr, endpoint) = match runtime::binding::<RpcComponent<R>>() {
156        Binding::Http(addr, endpoint) => (addr, endpoint),
157        _ => panic!("RPC component has non-HTTP binding"),
158    };
159
160    let path = format!("/{}/rpc", label);
161    let app = axum::Router::new().route(
162        &path,
163        axum::routing::post(move |axum::Json(req): axum::Json<R::Request>| {
164            let inner = inner.clone();
165            async move {
166                log::debug!("{} received RPC request: {}", label, req.verb());
167                let resp = inner.wait().await.handle(req).await;
168                axum::Json(resp)
169            }
170        }),
171    );
172
173    let listener = tokio::net::TcpListener::bind(addr).await.unwrap();
174    log::info!("{} listening on {}{}", label, endpoint, path);
175    axum::serve(listener, app).await.unwrap();
176}

178/// The HTTP client for making RPC calls.
179pub struct RpcHttpClient<R> {
180    endpoint: String,
181    client: reqwest::Client,
182    _marker: PhantomData<R>,
183}
184
185impl<R: Rpc> Clone for RpcHttpClient<R> {
186    fn clone(&self) -> Self {
187        RpcHttpClient {
188            endpoint: self.endpoint.clone(),
189            client: self.client.clone(),
190            _marker: PhantomData,
191        }
192    }
193}
194
195impl<R: Rpc> RpcHttpClient<R> {
196    fn new() -> RpcHttpClient<R> {
197        let endpoint = match runtime::binding::<RpcComponent<R>>() {
198            Binding::Http(_, endpoint) => endpoint,
199            _ => panic!("RPC component has non-HTTP binding"),
200        };
201        log::debug!(
202            "created HTTP RPC client for {}: {}",
203            runtime::label::<RpcComponent<R>>(),
204            endpoint,
205        );
206        RpcHttpClient {
207            endpoint,
208            client: reqwest::Client::new(),
209            _marker: PhantomData,
210        }
211    }
212
213    async fn call(&self, q: R::Request) -> Result<R::Response, RpcError> {
214        let label = runtime::label::<RpcComponent<R>>();
215        let url = format!("{}/{}/rpc", self.endpoint, label);
216        let resp = self
217            .client
218            .post(&url)
219            .json(&q)
220            .send()
221            .await
222            .map_err(|e| RpcError::Misc(format!("failed to send request: {}", e)))?;
223        let status = resp.status();
224        if !status.is_success() {
225            return Err(RpcError::Misc(format!(
226                "request failed with status code {}",
227                status
228            )));
229        }
230        let resp_msg = resp
231            .json::<R::Response>()
232            .await
233            .map_err(|e| RpcError::Misc(format!("failed to parse response: {}", e)))?;
234        Ok(resp_msg)
235    }
