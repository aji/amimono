1//! Subsystem for building components with an RPC interface.
2//!
3//! It's recommended to use this module via the [`rpc_ops!`][crate::rpc_ops]
4//! macro. When using that macro, it's rarely necessary to use any of the
5//! definitions in this module directly, however they are documented for the
6//! sake of completeness.
7
8use std::{
9    marker::PhantomData,
10    sync::{Arc, LazyLock},
11};
12
13use serde::{Deserialize, Serialize};
14use tokio::sync::SetOnce;
15
16use crate::{
17    config::{BindingType, ComponentConfig},
18    runtime::{self, Component},
19};

21/// A value that can be used as an RPC request or response.
22///
23/// Message types created by the [`rpc_ops!`][crate::rpc_ops] macro are
24/// automatically given an `RpcMessage` impl.
25pub trait RpcMessage: Serialize + for<'a> Deserialize<'a> + Send + 'static {
26    fn verb(&self) -> &'static str;
27}
28
29/// A value that can be used as an RPC request handler.
30///
31/// The `Instance` struct created by the [`rpc_ops!`][crate::rpc_ops] macro is
32/// automatically given an `Rpc` impl.
33pub trait Rpc: Sync + Send + 'static {
34    type Request: RpcMessage;
35    type Response: RpcMessage;
36
37    fn start() -> impl Future<Output = Self> + Send;
38
39    fn handle(&self, q: Self::Request) -> impl Future<Output = Self::Response> + Send;
40}
41
42type RpcInstance<R> = Arc<SetOnce<R>>;
43
44/// An RPC component, parameterized by an `Rpc` impl.
45///
46/// This type is an implementation detail that you should not need to use
47/// directly. The `ComponentConfig` returned by [`component`] is keyed by this
48/// type.
49pub struct RpcComponent<R>(PhantomData<R>);
50
51impl<R: Rpc> Component for RpcComponent<R> {
52    type Instance = RpcInstance<R>;
53}
54
55impl<R: Rpc> RpcComponent<R> {
56    #[tokio::main]
57    async fn entry() {
58        let instance = Arc::new(SetOnce::new());
59        runtime::set_instance::<Self>(instance.clone());
60        // we must call set_instance() asap, because get_instance::<T> blocks
61        // until the corresponding set_instance::<T> is called and we do not
62        // want to block in start() impls that make RPC calls.
63        instance.set(R::start().await).ok().unwrap();
64    }
65
66    fn component(label: String) -> ComponentConfig {
67        ComponentConfig {
68            label,
69            id: RpcComponent::<R>::id(),
70            binding: BindingType::Http,
71            entry: Self::entry,
72        }
73    }
74}
75
76/// Create a `ComponentConfig` for an `Rpc` impl.
77///
78/// When using the [`rpc_ops!`][crate::rpc_ops] macro, you should instead use
79/// the `component` function defined by that macro.
80pub fn component<R: Rpc>(label: String) -> ComponentConfig {
81    RpcComponent::<R>::component(label)
82}

84/// An error when making an RPC call.
85#[derive(Debug, Clone)]
86pub enum RpcError {
87    /// A miscellaneous error with an unstructured string message. These should
88    /// generally be assumed to be unrecoverable.
89    Misc(String),
90}
91
92/// A client for making requests to an RPC component.
93///
94/// Cloning values of this type will result in clients that share resources
95/// such as connection pools.
96///
97/// The `Client` struct defined by the [`rpc_ops!`][crate::rpc_ops] macro is a
98/// thin wrapper around this type.
99pub enum RpcClient<R> {
100    Local(LazyLock<RpcInstance<R>>),
101}
102
103impl<R: Rpc> Clone for RpcClient<R> {
104    fn clone(&self) -> Self {
105        RpcClient::<R>::new()
106    }
107}
108
109impl<R: Rpc> RpcClient<R> {
110    /// Create a new client for a particular `Rpc` impl. If an existing client
111    /// can be cloned, that should be preferred, as it will result in resources
112    /// being shared between the clients.
113    pub fn new() -> RpcClient<R> {
114        RpcClient::Local(LazyLock::new(|| {
115            runtime::get_instance::<RpcComponent<R>>().clone()
116        }))
117    }
118
119    /// Send a request. If the target `Rpc` impl belongs to a component that is
120    /// running in the same process, this will result in the target handler
121    /// being invoked directly.
122    pub async fn call(&self, q: R::Request) -> Result<R::Response, RpcError> {
123        match self {
124            RpcClient::Local(instance) => Ok(instance.wait().await.handle(q).await),
125        }
126    }
127
128    /// Returns a reference to the underlying `Rpc` impl if the target component
129    /// is running in the same process.
130    pub async fn local(&self) -> Option<&R> {
131        match self {
132            RpcClient::Local(instance) => Some(instance.wait().await),
133        }
134    }
135}
